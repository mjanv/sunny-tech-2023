Pipelines

        
        defmodule Maths do
            def oneliner(n) do
                Enum.sum(
                    Enum.map(
                        Enum.filter(
                            1..n, 
                            fn x -> Integer.is_odd(x) end
                        ), 
                        fn x -> x * x end
                    )
                )
            end

            def pipeline(n) do
                1..n
                |> Enum.filter(fn x -> Integer.is_odd(x) end)
                |> Enum.map(fn x -> x * x end)
                |> Enum.sum()
            end
        end 
        
    
        
            iex> Maths.oneliner(100)
            338350
            
            iex> Maths.pipeline(100)
            338350

        defmodule Maths do
            def sum([]) do
                0.0
            end 

            def sum([%Item{price: price} | items]) do
                price + sum(items)
            end

            def sum([head | tail]) when is_float(head) do
                head + sum(tail)
            end
        end
        
    
        
            iex> Maths.sum([
                    %Item{price: 2.1}, 
                    %Item{price: 3.4}, 
                    %Item{price: 1.0}
                ])
            6.5
        
            iex> Maths.sum([1.0, 2.0, 3.0])
            6.0
